CREATE VIEW [dbo].[vw_product]
AS
SELECT dbo.tbl_product.pid, dbo.tbl_product.product_id, dbo.tbl_category.pid AS pid_category, dbo.tbl_product.category_code, dbo.tbl_subcategory.pid AS pid_subcategory, dbo.tbl_product.subcategory_code, dbo.tbl_brand.pid AS pid_brand, dbo.tbl_product.brand_code, 
             dbo.tbl_product.product_name, dbo.tbl_product.product_desc, dbo.tbl_product.product_url, dbo.tbl_product.product_size, dbo.tbl_product.product_price
FROM   dbo.tbl_category INNER JOIN
             dbo.tbl_product ON dbo.tbl_category.category_code = dbo.tbl_product.category_code INNER JOIN
             dbo.tbl_subcategory ON dbo.tbl_product.subcategory_code = dbo.tbl_subcategory.subcategory_code INNER JOIN
             dbo.tbl_brand ON dbo.tbl_product.brand_code = dbo.tbl_brand.brand_code
